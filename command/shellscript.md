# 変数
## 宣言
	var=[数値]
	var='[文字列]'
	var=`[lsなどのコマンドのみ]`
	var=$([lsなどのコマンドのみ])
	var="$[varに設定する変数名]"
	var="${[varに設定する変数名]1}"
## 参照
	echo $var (変数を表示)
	echo ${var}1 (変数と1を表示)
	echo ${var}${var} (2つの変数を表示)
## メモ
先頭にlocalをつけるとローカル変数の宣言が可能。
先頭にreadonlyをつけると再定義不可。
=を扱うときは両脇にスペースを置かない。


# 配列
	[配列名] = ('[配列要素名]' '[配列要素名]')
	[配列名][[インデックス番号]]='[配列要素名]': 要素の追加。
	unset [配列名][[インデックス番号]]: 要素の削除。
	echo "${配列名[@]}": 配列の中身全て表示。
	echo "${配列名[[インデックス番号]]}": 配列の番号に応じた要素を表示。
	echo "${!配列名[@]}": 配列内の要素それぞれのインデックス番号を表示。
	echo "${#配列名[@]}": 配列内の要素の数を表示。

# 引数
## メモ
	シェルスクリプトを実行する際に、「./[ファイル名].sh [第一引数の値] [第二引数の値]」として引数を渡す。
	echo $0: ファイル名
	echo $1 $2 $3: 第1,2,3引数を表示。
	echo $@: 全引数を表示。
	echo $#: 引数の個数を表示。
	echo $0: ファイル名。
	echo $?: 実行したコマンドが成功した場合、0, 失敗の場合はそれ以外を返す。

# 入力
## 例
	read var
	echo $var
	read -p '[メッセージ]' var
## メモ
バッシュファイルを実行すると入力を受け付けるようになるので、そのまま入力すると、varに値が格納される。
## オプション
	-p: 入力を受け付けた際、メッセージを表示できる。
	-sp: メッセージを表示するが、自分の入力内容は出力しない。

# 出力
## 例
	read var
	echo var > text.txt
	echo var >> text.txt (追記)

# if文
## 例(条件文が1つ)
	if [ [条件文] ];
	then
		[条件文=Trueの処理]
	elif [ [第二条件文] ];
	then
		[第二条件の処理]
	else
		[その他の処理]
	fi
## 例(条件文が2つ)
	if [ [条件文] ] && [ [条件文] ];
	if [ [条件文] ] -a [ [条件文] ];
	if [ [条件文] ] || [ [条件文] ];
	if [ [条件文] ] -o [ [条件文] ];
## 例(否定)
	if [ [! 条件文] ];
## 例(短縮)
	echo 'A' && echo 'B' (左辺の処理が正しく実行されたら右辺の処理をする)
	echo 'A' || echo 'B' (左辺の処理に失敗したら右辺の処理をする)
## 文字比較
	=: c言語で言う ==
## 数値比較
 -eq: =
 -ne: !=
 -lt: <
 -le: <=
 -gt: >
 -ge: >=
 ## メモ
shellscriptでは比較する処理において、記号は扱わない。

# 四則演算
## 例
	echo $(( [演算内容] ))
## 変数使用var.
	echo $(( var1 + var2 ))

# 応用演算
## 準備
	yum install bc
## 例
	echo "20.5 + 4" | bc
	echo "$1 + $2" | bc (第一引数と第二引数を使用して計算)
	echo "scale=10;2/3" | bc -l (小数点以下10桁まで表示)
	echo "sqrt(2)" | bc -l (2の平方根を求めてくれる)
## bcオプション
	-l: 小数点以下の桁数指定など高度な計算が可能。
## メモ
bcコマンドを使用することで小数点以下の桁数を指定して、数値を扱える。
また、平方根、三角関数など様々な数学関数を計算できる。
計算内容を文字列として扱い、|を使ってbcコマンドに渡すと計算する。

# case文
## 例
	var=$1
	case $var in
		'blue' )
			echo "GO";;
		'red' )
			echo "STOP";;
		'yellow' )
			echo "WAIT";;
		[a-z] )
			echo "a~zまでの値です";;
		[A-Z] )
		echo "A~Zまでの値です";;
		[0-9] )
		echo "0~9までの値です";;
		*)
			echo "Wrong Color";;
	esac

# trap (プログラム終了時の挙動を設定)
## 例
```
	trap "[プログラム終了時に実行されるコマンド]"
	trap "[プログラム終了時に実行されるコマンド]" 15 (kill -15でのプロセス終了時に実行されるコマンド)
	trap "[プログラム終了時に実行されるコマンド]" 0 2 (終了時、割り込み終了時の時のみコマンド実行)
```
## メモ
- 1: 再起動
- 2: 割り込み(ctrl+c)
- 9: 強制停止(強制的に停止させているためtrapは設定できない)
- 15: プロセスの終了時

# デバッグモード(実行した際に現在処理しているコマンドや変数を確認できる)
## 実行方法
```
	bash -x ./script.sh
```
## メモ
set -xでデバッグモード
set +xでデフォルトとしてプログラム内で動的に変更できる。
またスクリプト内に#!/bin/bash -xでも可。
